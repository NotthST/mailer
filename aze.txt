#!/bin/bash

# Define your array of search strings
patterns=("password" "api_key" "secret_token" "confidential")

# Loop through each pattern and search recursively
for pattern in "${patterns[@]}"; do
  echo "ğŸ” Searching for: '$pattern'"
  grep -RInI --color=always "$pattern" . || echo "No matches for '$pattern'"
  echo "------------------------------------"
done
